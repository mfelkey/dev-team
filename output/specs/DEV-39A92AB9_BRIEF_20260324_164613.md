### Components

#### `app/page.tsx`
```tsx
import { useState } from "react";
import Header from "@/components/Header";
import HeroSection from "@/components/HeroSection";
import SportsCoverageStrip from "@/components/SportsCoverageStrip";
import SignalCardPreview from "@/components/SignalCardPreview";
import Footer from "@/components/Footer";

const LandingPage = () => {
  const [email, setEmail] = useState('');
  const [ageConfirmed, setAgeConfirmed] = useState(false);
  const [status, setStatus] = useState('');

  const handleSignupSubmit = async (e) => {
    e.preventDefault();
    try {
      await fetch('/api/signup', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, age_confirmed: ageConfirmed })
      });
      setStatus('You\'re on the list.');
    } catch (error) {
      setStatus('Error signing up. Try again later.');
    }
  };

  return (
    <div className="bg-[var(--color-navy)] text-white">
      <Header />
      <HeroSection status={status} setStatus={setStatus} handleSubmit={handleSignupSubmit} />
      <SportsCoverageStrip />
      <SignalCardPreview />
      <Footer />
    </div>
  );
};

export default LandingPage;
```

#### `app/api/signup/route.ts`
```ts
import type { NextApiRequest, NextApiResponse } from 'next';
import axios from 'axios';

const handleSignup = async (req: NextApiRequest, res: NextApiResponse) => {
  if (!req.body.email || !req.body.age_confirmed) return res.status(400).json({ error: 'Invalid data' });

  const { email, age_confirmed } = req.body;

  try {
    // Insert into PostgreSQL
    await axios.post(process.env.DATABASE_PUBLIC_URL + '/users', { email, ageConfirmed: age_confirmed });
    
    // Forward to Resend API for nurture sequence
    await axios.post('https://api.resend.com/emails', {
      from: 'signup@parallaxedge.com',
      to: [email],
      subject: "Welcome to ParallaxEdge Early Access!",
      html: '<p>You are now part of our early access list!</p>'
    }, { headers: { Authorization: `Bearer ${process.env.RESEND_API_KEY}` } });

    res.status(201).json({ success: true });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export default handleSignup;
```
